def calcular_serie(n):
    soma = 0.0
    for i in range(n):
        termo = 1 / (2 * i + 1)
        if i % 2 == 0:
            soma += termo
        else:
            soma -= termo
    return soma * 4

resultado = calcular_serie(10000)
print(resultado)
